USE MYWEB;

CREATE TABLE  MESSAGE (
	 ID   INT  NOT NULL AUTO_INCREMENT PRIMARY KEY,
	 TITLE  VARCHAR(500) NULL,
	 REMARK  VARCHAR(4000) NULL,	
	 CREATE_TIME   DATETIME  NULL
);

CREATE TABLE  MESSAGE_RE (
	 ID   INT  NOT NULL AUTO_INCREMENT PRIMARY KEY,
	 MSG_ID INT  NOT NULL,
	 CREATE_TIME   DATETIME  NULL
);
